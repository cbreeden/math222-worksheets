\newproblem{deforindef:set1}{% 
For each of the following, state whether the object is a function or a number.  If it is a function, state what variable it is a function of.
\begin{enumerate}
\item $\int_1^x e^{\cos(t)} dt$
\item $\int_1^3 \sin(s) ds$
\item $\int \ln(x) dx$
\item $\int_t^{t^3} \ln(\cos(x))dx$
\end{enumerate}}{% 
\begin{enumerate}
\item This is a function of $x$.
\item This is a number.
\item This is a function of $x$ (one of many really).
\item This is a function of $t$.
\end{enumerate}
}
